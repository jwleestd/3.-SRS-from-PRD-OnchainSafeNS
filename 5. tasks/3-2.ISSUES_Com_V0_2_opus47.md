```typescript
// middleware.ts (프로젝트 루트)
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

export function middleware(req: NextRequest) {
  const { pathname } = req.nextUrl;
  const adminSession = req.cookies.get('admin_session')?.value;
  const isAuthenticated = adminSession === process.env.ADMIN_PASSWORD;

  // Admin 페이지 보호: /(admin)/* → /auth/admin-login 리다이렉트
  if (pathname.startsWith('/admin')) {
    if (!isAuthenticated) {
      const loginUrl = new URL('/auth/admin-login', req.url);
      loginUrl.searchParams.set('redirect', pathname);
      return NextResponse.redirect(loginUrl);
    }
  }

  // Admin API 보호: /api/v1/admin/* → 401 JSON
  if (pathname.startsWith('/api/v1/admin')) {
    if (!isAuthenticated) {
      return NextResponse.json(
        {
          success: false,
          error: { code: 'UNAUTHORIZED', message: '인증이 필요합니다' },
          meta: { timestamp: new Date().toISOString(), request_id: crypto.randomUUID() }
        },
        { status: 401 }
      );
    }
  }

  return NextResponse.next();
}

export const config = {
  matcher: ['/admin/:path*', '/api/v1/admin/:path*']
};
```

### CMD-AUTH-001 — 이메일 인증 코드 발송

```typescript
// app/api/v1/auth/verify-email/route.ts (POST handler — send 분기)
import { NextRequest, NextResponse } from 'next/server';
import { prisma } from '@/lib/prisma';
import { sendVerificationCode } from '@/lib/email';
import { validateEmail } from '@/lib/validators';

export async function POST(req: NextRequest) {
  const body = await req.json();
  const { email, action } = body;

  // 1. 이메일 형식 검증
  if (!validateEmail(email)) {
    return NextResponse.json(
      { success: false, error: { code: 'VALIDATION_ERROR', message: '올바른 이메일 형식을 입력하세요' } },
      { status: 400 }
    );
  }

  if (action === 'send') {
    // 2. 1분 이내 재요청 차단
    const existing = await prisma.user.findUnique({ where: { email } });
    if (existing?.verification_expires_at) {
      const remainingMs = existing.verification_expires_at.getTime() - Date.now();
      if (remainingMs > 9 * 60 * 1000) {  // 9분+ 남음 = 1분 미만 경과
        return NextResponse.json(
          { success: false, error: { code: 'RATE_LIMITED', message: '잠시 후 다시 시도하세요' } },
          { status: 429 }
        );
      }
    }

    // 3. 6자리 코드 생성 + DB 저장 + Resend 발송
    const code = String(Math.floor(100000 + Math.random() * 900000));
    const expiresAt = new Date(Date.now() + 10 * 60 * 1000);  // 10분 후

    await prisma.user.upsert({
      where: { email },
      update: { verification_code: code, verification_expires_at: expiresAt },
      create: { email, verification_code: code, verification_expires_at: expiresAt, email_verified: false }
    });

    await sendVerificationCode(email, code);  // 비차단 처리

    return NextResponse.json(
      { success: true, data: { message: '코드가 발송되었습니다', expires_in_sec: 600 } },
      { status: 200 }
    );
  }
  // action === 'verify' 분기는 CMD-AUTH-002에서 처리
}
```

### CMD-AUTH-002 — 인증 코드 확인

```typescript
// app/api/v1/auth/verify-email/route.ts (POST handler — verify 분기)
// 위 CMD-AUTH-001 같은 route 내 action='verify' 분기

if (action === 'verify') {
  const { email, code } = body;

  // 1. 코드 형식 검증
  if (!/^\d{6}$/.test(code)) {
    return NextResponse.json(
      { success: false, error: { code: 'VALIDATION_ERROR', message: '6자리 코드를 입력하세요' } },
      { status: 400 }
    );
  }

  // 2. USER 조회
  const user = await prisma.user.findUnique({ where: { email } });
  if (!user || !user.verification_code) {
    return NextResponse.json(
      { success: false, error: { code: 'INVALID_CODE', message: '인증 코드를 먼저 발송하세요' } },
      { status: 401 }
    );
  }

  // 3. 코드 만료 확인
  if (user.verification_expires_at && user.verification_expires_at < new Date()) {
    return NextResponse.json(
      { success: false, error: { code: 'EXPIRED', message: '코드가 만료되었습니다' } },
      { status: 410 }
    );
  }

  // 4. 코드 일치 확인
  if (user.verification_code !== code) {
    return NextResponse.json(
      { success: false, error: { code: 'INVALID_CODE', message: '코드가 일치하지 않습니다' } },
      { status: 401 }
    );
  }

  // 5. email_verified=true + 코드 무효화
  await prisma.user.update({
    where: { user_id: user.user_id },
    data: {
      email_verified: true,
      verification_code: null,
      verification_expires_at: null
    }
  });

  // 6. 쿠키 설정 + 응답
  const response = NextResponse.json(
    { success: true, data: { user_id: user.user_id, email_verified: true } },
    { status: 200 }
  );
  response.cookies.set('user_session', user.user_id, {
    httpOnly: true,
    secure: true,
    sameSite: 'lax',
    maxAge: 7 * 24 * 60 * 60  // 7일
  });
  return response;
}
```

```typescript
// app/api/v1/fraud/lookup/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { prisma } from '@/lib/prisma';
import { validateAddress } from '@/lib/validators';
import { SUPPORTED_CHAINS } from '@/lib/constants';

export async function GET(req: NextRequest) {
  const { searchParams } = new URL(req.url);
  const address = searchParams.get('address');
  const chain = searchParams.get('chain');

  // 1. 파라미터 누락
  if (!address || !chain) {
    return NextResponse.json(
      { success: false, error: { code: 'VALIDATION_ERROR', message: '주소와 체인을 모두 입력하세요' } },
      { status: 400 }
    );
  }

  // 2. 미지원 체인
  if (!SUPPORTED_CHAINS.includes(chain as any)) {
    return NextResponse.json(
      { success: false, error: { code: 'VALIDATION_ERROR', message: '지원하지 않는 체인입니다' } },
      { status: 400 }
    );
  }

  // 3. 주소 형식 검증
  if (!validateAddress(address, chain)) {
    return NextResponse.json(
      { success: false, error: { code: 'VALIDATION_ERROR', message: '올바른 주소 형식을 입력하세요' } },
      { status: 400 }
    );
  }

  // 4. FRAUD_ADDRESS 조회 (인덱스 활용)
  const fraud = await prisma.fraudAddress.findFirst({
    where: { address: address.toLowerCase(), chain }
  });

  if (!fraud) {
    return NextResponse.json({
      success: true,
      data: { found: false, message: '등록된 사기 이력이 없습니다' }
    });
  }

  return NextResponse.json({
    success: true,
    data: {
      found: true,
      risk_level: fraud.risk_level,
      report_count: fraud.report_count,
      source_type: fraud.source_type ?? '출처 미확인'  // QRY-FRAUD-003 폴백
    }
  });
}
```

```typescript
// lib/validators.ts 의 validateAddress 활용

// lib/validators.ts (LIB-003)
import { ADDRESS_PATTERNS } from './constants';

export function validateAddress(address: string, chain: string): boolean {
  const pattern = ADDRESS_PATTERNS[chain];
  if (!pattern) return false;
  return pattern.test(address);
}

// lib/constants.ts (LIB-004)
export const ADDRESS_PATTERNS: Record<string, RegExp> = {
  ethereum: /^0x[a-fA-F0-9]{40}$/,
  polygon:  /^0x[a-fA-F0-9]{40}$/,
  bsc:      /^0x[a-fA-F0-9]{40}$/,
  solana:   /^[1-9A-HJ-NP-Za-km-z]{32,44}$/
};

export const SUPPORTED_CHAINS = ['ethereum', 'polygon', 'bsc', 'solana'] as const;
```

```typescript
// app/api/v1/fraud/report/route.ts (POST)
import { NextRequest, NextResponse } from 'next/server';
import { prisma } from '@/lib/prisma';
import { validateAddress } from '@/lib/validators';
import { sendReportConfirmation } from '@/lib/email';

export async function POST(req: NextRequest) {
  // 1. 인증 확인
  const userId = req.cookies.get('user_session')?.value;
  if (!userId) {
    return NextResponse.json(
      { success: false, error: { code: 'UNAUTHORIZED', message: '인증이 필요합니다' } },
      { status: 401 }
    );
  }

  const user = await prisma.user.findUnique({ where: { user_id: userId } });
  if (!user || !user.email_verified) {
    return NextResponse.json(
      { success: false, error: { code: 'UNAUTHORIZED', message: '이메일 인증이 필요합니다' } },
      { status: 401 }
    );
  }

  // 2. 제한 유저 확인 (CMD-FRAUD-004)
  if (user.report_restriction_until && user.report_restriction_until > new Date()) {
    return NextResponse.json(
      {
        success: false,
        error: {
          code: 'RATE_LIMITED',
          message: `신고 제한 중입니다. ${user.report_restriction_until.toISOString().split('T')[0]}까지`
        }
      },
      { status: 429 }
    );
  }

  // 3. 입력 검증
  const body = await req.json();
  const { reported_address, chain, description, evidence_url } = body;

  if (!validateAddress(reported_address, chain)) {
    return NextResponse.json(
      { success: false, error: { code: 'VALIDATION_ERROR', message: '올바른 주소 형식을 입력하세요' } },
      { status: 400 }
    );
  }
  if (!description || description.length < 10 || description.length > 1000) {
    return NextResponse.json(
      { success: false, error: { code: 'VALIDATION_ERROR', message: '피해 내역은 10~1000자' } },
      { status: 400 }
    );
  }

  // 4. 중복 신고 확인 (CMD-FRAUD-002)
  const existing = await prisma.fraudReport.findFirst({
    where: { reporter_id: userId, reported_address: reported_address.toLowerCase() }
  });
  if (existing) {
    return NextResponse.json(
      { success: false, error: { code: 'CONFLICT', message: '이미 신고한 주소입니다' } },
      { status: 409 }
    );
  }

  // 5. FRAUD_REPORT 생성
  const report = await prisma.fraudReport.create({
    data: {
      reporter_id: userId,
      reported_address: reported_address.toLowerCase(),
      chain,
      description,
      evidence_url: evidence_url || null,
      status: 'submitted'
    }
  });

  // 6. FRAUD_ADDRESS 기존 존재 시 report_count 증가 (CMD-FRAUD-002)
  await prisma.fraudAddress.updateMany({
    where: { address: reported_address.toLowerCase(), chain },
    data: { report_count: { increment: 1 } }
  });

  // 7. 이메일 발송 (비차단 — CMD-FRAUD-003)
  sendReportConfirmation({
    email: user.email,
    report_id: report.report_id,
    reported_address,
    chain,
    reported_at: report.reported_at
  }).catch((err) => console.error('[Email] sendReportConfirmation failed:', err));

  return NextResponse.json(
    { success: true, data: { report_id: report.report_id, status: 'submitted' } },
    { status: 201 }
```

```typescript
export async function POST(req: NextRequest) {
  // 1. 인증 확인
  const userId = req.cookies.get('user_session')?.value;
  if (!userId) {
    return NextResponse.json(

      { status: 400 }
    );

  const existing = await prisma.safeName.findUnique({ where: { human_name: finalName } });
  if (existing) {
```
